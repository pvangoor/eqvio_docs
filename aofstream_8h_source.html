58  void flushStreamToFile() {
59  std::chrono::steady_clock::time_point lastFlush = std::chrono::steady_clock::now();
60  while (true) {
61  std::unique_lock lck(streamMutex);
62  buffer_cv.wait(lck, [this]() { return !(streamBuffer.rdbuf()->in_avail() == 0) || destructorCalled; });
63  outputFile << streamBuffer.rdbuf();
64 
65  // Flush every 5 seconds
66  if ((std::chrono::steady_clock::now() - lastFlush) > std::chrono::duration<double>(5.0)) {
67  lastFlush = std::chrono::steady_clock::now();
68  outputFile << std::flush;
69  }
70 
71  if (this->destructorCalled) {
72  break;
73  }
74  }
75  if (!(streamBuffer.rdbuf()->in_avail() == 0)) {
76  outputFile << streamBuffer.rdbuf();
77  }
78  outputFile << std::flush;
79  }

84  aofstream(const std::string& fname) {

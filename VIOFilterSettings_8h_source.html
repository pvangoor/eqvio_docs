34 static CoordinateChoice coordinateSelection(const YAML::Node& ccNode) {
35  std::string choice;
36  safeConfig(ccNode, "settings:coordinateChoice", choice);
37 
38  if (choice == "Euclidean")
39  return CoordinateChoice::Euclidean;
40  else if (choice == "InvDepth")
41  return CoordinateChoice::InvDepth;
42  else if (choice == "Normal")
43  return CoordinateChoice::Normal;
44  else
45  throw std::runtime_error("Invalid coordinate choice. Valid choices are Euclidean, InvDepth, Normal.");
46 }

176 inline Eigen::MatrixXd VIOFilter::Settings::constructStateGainMatrix(const size_t& numLandmarks) const {
177  Eigen::MatrixXd PMat = Eigen::MatrixXd::Identity(
178  VIOSensorState::CompDim + 3 * numLandmarks, VIOSensorState::CompDim + 3 * numLandmarks);
179  PMat.block<3, 3>(0, 0) *= this->biasOmegaProcessVariance;
180  PMat.block<3, 3>(3, 3) *= this->biasAccelProcessVariance;
181  PMat.block<3, 3>(6, 6) *= this->attitudeProcessVariance;
182  PMat.block<3, 3>(9, 9) *= this->positionProcessVariance;
183  PMat.block<3, 3>(12, 12) *= this->velocityProcessVariance;
184  PMat.block<3, 3>(15, 15) *= this->cameraAttitudeProcessVariance;
185  PMat.block<3, 3>(18, 18) *= this->cameraPositionProcessVariance;
186  PMat.block(VIOSensorState::CompDim, VIOSensorState::CompDim, 3 * numLandmarks, 3 * numLandmarks) *=
187  this->pointProcessVariance;
188 
189  return PMat;
190 }

192 inline Eigen::Matrix<double, IMUVelocity::CompDim, IMUVelocity::CompDim>
193 VIOFilter::Settings::constructInputGainMatrix() const {
194  Eigen::Matrix<double, IMUVelocity::CompDim, IMUVelocity::CompDim> R =
195  Eigen::Matrix<double, IMUVelocity::CompDim, IMUVelocity::CompDim>::Identity();
196  R.block<3, 3>(0, 0) *= this->velGyrNoise * this->velGyrNoise;
197  R.block<3, 3>(3, 3) *= this->velAccNoise * this->velAccNoise;
198  R.block<3, 3>(6, 6) *= this->velGyrBiasWalk * this->velGyrBiasWalk;
199  R.block<3, 3>(9, 9) *= this->velAccBiasWalk * this->velAccBiasWalk;
200  return R;
201 }

208 inline Eigen::MatrixXd VIOFilter::Settings::constructInitialStateCovariance(const size_t& numLandmarks) const {
209  Eigen::MatrixXd Sigma(VIOSensorState::CompDim + 3 * numLandmarks, VIOSensorState::CompDim + 3 * numLandmarks);
210  Sigma.setIdentity();
211  Sigma.block<3, 3>(0, 0) *= this->initialBiasOmegaVariance;
212  Sigma.block<3, 3>(3, 3) *= this->initialBiasAccelVariance;
213  Sigma.block<3, 3>(6, 6) *= this->initialAttitudeVariance;
214  Sigma.block<3, 3>(9, 9) *= this->initialPositionVariance;
215  Sigma.block<3, 3>(12, 12) *= this->initialVelocityVariance;
216  Sigma.block<3, 3>(15, 15) *= this->initialCameraAttitudeVariance;
217  Sigma.block<3, 3>(18, 18) *= this->initialCameraPositionVariance;
218 
219  Sigma.block(VIOSensorState::CompDim, VIOSensorState::CompDim, 3 * numLandmarks, 3 * numLandmarks) *=
220  this->initialPointVariance;
221 
222  if (initialPointDepthVariance > 0) {
223  for (size_t i = 0; i < numLandmarks; ++i) {
224  Sigma(VIOSensorState::CompDim + 3 * i + 2, VIOSensorState::CompDim + 3 * i + 2) = initialPointDepthVariance;
225  }
226  }
227 
228  return Sigma;
229 }

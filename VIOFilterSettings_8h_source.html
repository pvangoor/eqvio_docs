32 enum class CoordinateChoice { Euclidean, InvDepth, Normal };

38 static CoordinateChoice coordinateSelection(const YAML::Node& ccNode) {
39  std::string choice;
40  safeConfig(ccNode, "settings:coordinateChoice", choice);
41 
42  if (choice == "Euclidean")
43  return CoordinateChoice::Euclidean;
44  else if (choice == "InvDepth")
45  return CoordinateChoice::InvDepth;
46  else if (choice == "Normal")
47  return CoordinateChoice::Normal;
48  else
49  throw std::runtime_error("Invalid coordinate choice. Valid choices are Euclidean, InvDepth, Normal.");
50 }
